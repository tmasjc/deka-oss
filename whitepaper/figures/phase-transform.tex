% F2 — the four-phase precision/cost/legibility transformation
\begin{tikzpicture}[
  font=\small,
  node distance=4mm,
  phase/.style={draw, rounded corners, align=center, text width=38mm, minimum height=10mm, fill=RoyalBlue!5},
  note/.style={align=left, text width=92mm, font=\footnotesize},
  art/.style={align=center, font=\footnotesize\itshape, text=black!55},
  arr/.style={-{Stealth}, thick}
]
\node[art] (a0) {fuzzy operator intuition};
\node[phase, below=of a0] (p1) {\textbf{1. Probe}};
\node[note, right=7mm of p1] (n1) {Tune hybrid search while the operator rates chunks. \emph{Transforms} intuition into human-validated FIT / NOT\_FIT examples.};
\node[art, below=of p1] (a1) {rated examples};
\node[phase, below=of a1] (p2) {\textbf{2. Harvest}};
\node[note, right=7mm of p2] (n2) {The FITs become a query set; sweep the corpus by embedding distance. \emph{Transforms} examples into a geometric cohort.};
\node[art, below=of p2] (a2) {geometric cohort};
\node[phase, below=of a2] (p3) {\textbf{3. Refine}};
\node[note, right=7mm of p3] (n3) {Derive a rubric prompt from the session; judge a stratified sample. \emph{Transforms} geometry into a readable, auditable language boundary.};
\node[art, below=of p3] (a3) {language rubric};
\node[phase, below=of a3] (p4) {\textbf{4. Apply}};
\node[note, right=7mm of p4] (n4) {Train a cheap classifier on the judged sample; score the full cohort. \emph{Transforms} expensive LLM judging into near-zero-cost labels at scale.};
\node[art, below=of p4] (a4) {labelled cohort at scale};
